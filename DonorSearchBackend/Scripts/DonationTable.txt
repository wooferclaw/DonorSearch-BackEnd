CREATE TABLE public."Donations"
(
    id integer NOT NULL DEFAULT nextval('"Donations_id_seq"'::regclass),
    blood_class_ids integer NOT NULL,
    "donation_Id" integer,
    succeed boolean,
    recomendation_timestamp timestamp without time zone NOT NULL,
    vk_id integer NOT NULL,
    donation_timestamp timestamp without time zone NOT NULL,
    station_id integer NOT NULL,
    status_id integer NOT NULL,
    CONSTRAINT "PK_Donations" PRIMARY KEY (id)
)
WITH (
    OIDS = FALSE
)
TABLESPACE pg_default;

    OWNER to postgres;
