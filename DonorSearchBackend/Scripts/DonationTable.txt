CREATE TABLE public."Donations"
(
    "Id" integer NOT NULL DEFAULT nextval('"Donations_Id_seq"'::regclass),
    "VkId" integer NOT NULL,
    "DonationTimestamp" timestamp without time zone NOT NULL,
    "StationId" integer NOT NULL,
    "StatusId" integer NOT NULL,
    "DonationType" integer NOT NULL,
    "DonationId" integer NOT NULL,
    "Succeed" boolean NOT NULL,
    "RecomendationTimestamp" timestamp without time zone NOT NULL,
    CONSTRAINT "PK_Donations" PRIMARY KEY ("Id")
)
WITH (
    OIDS = FALSE
)
TABLESPACE pg_default;

    OWNER to postgres;
