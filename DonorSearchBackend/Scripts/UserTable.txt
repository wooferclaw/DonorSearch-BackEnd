CREATE TABLE public."Users"
(
    "VkId" integer NOT NULL DEFAULT nextval('"Users_VkId_seq"'::regclass),
    "DSId" integer NOT NULL,
    CONSTRAINT "PK_Users" PRIMARY KEY ("VkId")
)
WITH (
    OIDS = FALSE
)
TABLESPACE pg_default;

ALTER TABLE public."Users"
    OWNER to postgres;

-- Index: IX_Users_DSId

CREATE UNIQUE INDEX "IX_Users_DSId"
    ON public."Users" USING btree
    ("DSId")
    TABLESPACE pg_default;
